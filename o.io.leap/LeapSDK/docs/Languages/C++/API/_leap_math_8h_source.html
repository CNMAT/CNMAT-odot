240  float angleTo(const Vector& other) const {
241  float denom = this->magnitudeSquared() * other.magnitudeSquared();

749  void setRotation(const Vector& axis, float angleRadians) {
750  const Vector n = axis.normalized();
751  const float s = std::sin(angleRadians);
752  const float c = std::cos(angleRadians);
753  const float C = (1-c);
754 
755  xBasis = Vector(n[0]*n[0]*C + c, n[0]*n[1]*C - n[2]*s, n[0]*n[2]*C + n[1]*s);
756  yBasis = Vector(n[1]*n[0]*C + n[2]*s, n[1]*n[1]*C + c, n[1]*n[2]*C - n[0]*s);
757  zBasis = Vector(n[2]*n[0]*C - n[1]*s, n[2]*n[1]*C + n[0]*s, n[2]*n[2]*C + c );
758  }

796  Matrix rigidInverse() const {
797  Matrix rotInverse = Matrix(Vector(xBasis[0], yBasis[0], zBasis[0]),
